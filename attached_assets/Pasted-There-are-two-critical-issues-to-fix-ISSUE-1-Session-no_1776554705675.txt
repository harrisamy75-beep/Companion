ISSUE 3 — Login screen must set session properly:
In the POST /auth/login route, make sure it:
1. Sets req.session.userId to the slugified name
2. Calls req.session.save() explicitly before responding:

router.post("/login", (req, res) => {
  const { userId } = req.body;
  if (!userId) { res.status(400).json({ error: "userId required" }); return; }
  const slug = userId.toLowerCase().replace(/\s+/g, '-');
  req.session.userId = slug;
  req.session.save((err) => {
    if (err) { res.status(500).json({ error: "Session save failed" }); return; }
    res.json({ ok: true, userId: slug });
  });
});

export async function api(path: string, options: RequestInit = {}) {
  const response = await fetch(path, {
    ...options,
    credentials: "include",
    headers: {
      "Content-Type": "application/json",
      ...options.headers,
    },
  });
  return response;
}
