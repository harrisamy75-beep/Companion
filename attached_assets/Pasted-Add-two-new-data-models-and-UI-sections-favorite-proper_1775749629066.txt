Step 1 — new table lib/db/src/schema/favorite-properties.ts:

export const favoritePropertiesTable = pgTable("favorite_properties", {
  id: serial("id").primaryKey(),
  userId: text("user_id").notNull(),
  propertyName: text("property_name").notNull(),   // "Rosewood Miramar"
  brand: text("brand"),                             // "Rosewood"
  location: text("location"),                       // "Montecito, CA"
  category: text("category"),                       // "hotel" | "resort" | "villa" | "restaurant"
  tier: text("tier"),                               // "loved" | "liked" | "avoid"
  tags: jsonb("tags").$type<string[]>().default([]), // what you loved: "service", "food", "design", "value"
  notes: text("notes"),                             // "kids loved the beach, ask for ocean suite"
  visitedAt: text("visited_at"),                    // freeform "Summer 2023"
  createdAt: timestamp("created_at", { withTimezone: true }).notNull().defaultNow(),
});

Step 2 — new table lib/db/src/schema/loyalty-programs.ts:

export const loyaltyProgramsTable = pgTable("loyalty_programs", {
  id: serial("id").primaryKey(),
  userId: text("user_id").notNull(),
  programName: text("program_name").notNull(),  // "World of Hyatt"
  brand: text("brand").notNull(),               // "Hyatt"
  membershipNumber: text("membership_number"),  // for auto-fill
  tier: text("tier"),                           // "Member" | "Discoverist" | "Explorist" | "Globalist"
  notes: text("notes"),
  createdAt: timestamp("created_at", { withTimezone: true }).notNull().defaultNow(),
});

Export both from lib/db/src/schema/index.ts.

Step 3 — new route artifacts/api-server/src/routes/properties.ts:

GET /properties — list all for userId, ordered by tier (loved first)
POST /properties — add one
PUT /properties/:id — edit
DELETE /properties/:id — delete

GET /properties/brands — return deduplicated list of brands 
from favoriteProperties for this user. Used to seed the 
review scoring context.
