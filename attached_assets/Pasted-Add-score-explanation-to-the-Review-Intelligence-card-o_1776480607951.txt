function buildMatchExplanation(breakdown, userTags, score) {
  const strengths = [];
  const gaps = [];
  
  if (breakdown.luxuryValue.avgScore >= 7) 
    strengths.push("strong luxury-value ratio");
  else if (breakdown.luxuryValue.avgScore <= 4)
    gaps.push("luxury-value doesn't match your tier");
    
  if (userTags.includes("foodie") && breakdown.foodie.avgScore >= 7)
    strengths.push("excellent food credentials");
  else if (userTags.includes("foodie") && breakdown.foodie.avgScore <= 4)
    gaps.push("food didn't impress reviewers");
    
  if (userTags.includes("eco") && breakdown.eco.avgScore >= 7)
    strengths.push("strong eco practices");
  else if (userTags.includes("eco") && breakdown.eco.avgScore <= 4)
    gaps.push("limited eco credentials");

  if (userTags.includes("adventurous_menu") && 
      breakdown.adventurousMenu.avgScore <= 4)
    gaps.push("menu plays it safe");

  let explanation = "";
  if (strengths.length > 0) 
    explanation += `Scores well for ${strengths.join(" and ")}. `;
  if (gaps.length > 0)
    explanation += `Held back by ${gaps.join(" and ")}.`;
  if (explanation === "") 
    explanation = `Solid mid-range match across your key criteria.`;
    
  return explanation;
}
